[01]+ {
    int num = 0, base = 2, exp;
    int result;
    int len = strlen(yytext);
    for (int i = 0; i < len; i++) {
        if (yytext[i] == '1') {
	    result = 1;
	    exp = len-i-1;
            while (exp != 0) {
        	result *= base;
        	--exp;
    	}
	    num += result;
        }
    }
    if (num == 0) {
        printf("The input is not a power of two\n");
    } else if ((num & (num-1)) == 0) {
        printf("The input is a power of two\n");
    } else {
        printf("The input is not a power of two\n");
    }
}

. {
    printf("Invalid input\n");
}
